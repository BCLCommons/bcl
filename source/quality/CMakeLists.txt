CMAKE_MINIMUM_REQUIRED( VERSION 2.8.8)

SET( LIBRARY_NAME bclquality)
SET( LIBRARY_OBJECTS ${LIBRARY_NAME}-objects)
PROJECT( lib${LIBRARY_NAME})

INCLUDE_DIRECTORIES( "${CMAKE_CURRENT_SOURCE_DIR}/../../include/")

SET(
  LIBRARY_SOURCES
  ${CMAKE_CURRENT_SOURCE_DIR}/bcl_quality.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/bcl_quality_average.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/bcl_quality_const_measure.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/bcl_quality_dme.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/bcl_quality_dmf.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/bcl_quality_gdt.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/bcl_quality_lcs.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/bcl_quality_maxsub.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/bcl_quality_measures.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/bcl_quality_rmsd.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/bcl_quality_rmsd_preprocessor.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/bcl_quality_superimpose_measures.cpp
)
MACRO_CHECK_CMAKE_DIRECTORY_LIST( ${CMAKE_CURRENT_SOURCE_DIR} "${LIBRARY_SOURCES}")

ADD_LIBRARY( ${LIBRARY_OBJECTS} OBJECT ${LIBRARY_SOURCES})

IF( LINK_NAMESPACE_ARCHIVES)
  ADD_LIBRARY( ${LIBRARY_NAME} STATIC $<TARGET_OBJECTS:${LIBRARY_OBJECTS}>)
ENDIF()

SET( SUBDIRECTORY_LIBRARY_NAME ${LIBRARY_NAME} PARENT_SCOPE)
SET( SUBDIRECTORY_LIBRARY_OBJECTS ${LIBRARY_OBJECTS} PARENT_SCOPE)
