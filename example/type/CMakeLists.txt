CMAKE_MINIMUM_REQUIRED( VERSION 2.8.8)

SET( LIBRARY_NAME bclexampletype)
SET( LIBRARY_OBJECTS ${LIBRARY_NAME}-objects)
PROJECT( lib${LIBRARY_NAME})

SET(
  LIBRARY_SOURCES
  ${CMAKE_CURRENT_SOURCE_DIR}/example_type.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/example_type_chooser.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/example_type_compare.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/example_type_enable_if.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/example_type_is_a.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/example_type_is_derived_from.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/example_type_is_map.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/example_type_is_sequence.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/example_type_remove_const_ref.cpp
)
MACRO_CHECK_CMAKE_DIRECTORY_LIST( ${CMAKE_CURRENT_SOURCE_DIR} "${LIBRARY_SOURCES}")

ADD_LIBRARY( ${LIBRARY_OBJECTS} OBJECT ${LIBRARY_SOURCES})

IF( LINK_NAMESPACE_ARCHIVES)
  ADD_LIBRARY( ${LIBRARY_NAME} STATIC $<TARGET_OBJECTS:${LIBRARY_OBJECTS}>)
ENDIF()

SET( SUBDIRECTORY_LIBRARY_NAME ${LIBRARY_NAME} PARENT_SCOPE)
SET( SUBDIRECTORY_LIBRARY_OBJECTS ${LIBRARY_OBJECTS} PARENT_SCOPE)
