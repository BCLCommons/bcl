CMAKE_MINIMUM_REQUIRED( VERSION 2.8.8)

SET( LIBRARY_NAME bclexamplestorage)
SET( LIBRARY_OBJECTS ${LIBRARY_NAME}-objects)
PROJECT( lib${LIBRARY_NAME})

SET(
  LIBRARY_SOURCES
  ${CMAKE_CURRENT_SOURCE_DIR}/example_storage.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/example_storage_hash_map.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/example_storage_list.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/example_storage_map.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/example_storage_object_nd_hash_map.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/example_storage_pair.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/example_storage_row.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/example_storage_set.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/example_storage_symmetric_matrix.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/example_storage_table.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/example_storage_table_header.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/example_storage_triplet.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/example_storage_vector.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/example_storage_vector_nd.cpp
)
MACRO_CHECK_CMAKE_DIRECTORY_LIST( ${CMAKE_CURRENT_SOURCE_DIR} "${LIBRARY_SOURCES}")

ADD_LIBRARY( ${LIBRARY_OBJECTS} OBJECT ${LIBRARY_SOURCES})

IF( LINK_NAMESPACE_ARCHIVES)
  ADD_LIBRARY( ${LIBRARY_NAME} STATIC $<TARGET_OBJECTS:${LIBRARY_OBJECTS}>)
ENDIF()

SET( SUBDIRECTORY_LIBRARY_NAME ${LIBRARY_NAME} PARENT_SCOPE)
SET( SUBDIRECTORY_LIBRARY_OBJECTS ${LIBRARY_OBJECTS} PARENT_SCOPE)
